function openActivityModal(){
  document.getElementById('act-date').value=new Date().toISOString().split('T')[0];
  document.getElementById('activity-modal').classList.add('open');
}

async function saveActivity(){
  const data={
    activity_name:    document.getElementById('act-name').value,
    activity_name_en: document.getElementById('act-name-en').value||null,
    activity_date:    document.getElementById('act-date').value,
    activity_type:    document.getElementById('act-type').value,
    emirati_led:      document.getElementById('act-emirati').value==='true'
  };
  if(!data.activity_name||!data.activity_date){ showToast('⚠️ أدخل الاسم والتاريخ','amber'); return; }
  try{
    await sbPost('national_activities',data);
    closeModal('activity-modal');
    await loadActivities();
    showToast('✅ تم إضافة النشاط');
  }catch(e){ showToast('❌ فشل الحفظ','red'); }
}

let currentAuthority = null;
let currentEmailDraft = null;
let schoolInfo = {};

async function initOfficialComms() {
  // استرجع بيانات المدرسة من app_settings
  try {
    const sets = await sbGet('app_settings', 'key=in.(school_name,school_esid,school_type,school_emirate,principal_name,school_phone,school_email)');
    if (sets && sets.length) {
      const sm = {};
      sets.forEach(s => sm[s.key] = s.value);
      schoolInfo = {
        name:      sm.school_name      || 'مدرسة الجود',
        id_number: sm.school_esid      || '1705',
        principal: sm.principal_name   || '',
        phone:     sm.school_phone     || '',
        email:     sm.school_email     || '',
        type:      sm.school_type      || 'government',
        emirate:   sm.school_emirate   || 'Abu Dhabi',
      };
    } else {
      schoolInfo = { name:'مدرسة الجود', id_number:'1705', type:'government', emirate:'Abu Dhabi', principal:'', phone:'', email:'' };
    }
  } catch(e) {
    schoolInfo = { name:'مدرسة الجود', id_number:'1705', type:'government', emirate:'Abu Dhabi', principal:'', phone:'', email:'' };
  }

  currentAuthority = getAuthorityLocal(schoolInfo.type, schoolInfo.emirate);
  renderAuthorityBadge();
  renderAuthorityChannels();
  renderRequestTypes();
  await loadCommsLog();
}

function getAuthorityLocal(schoolType, emirate) {
  const emirateMap = {
    'Abu Dhabi':      { government: 'government_abudhabi', private: 'private_abudhabi' },
    'Dubai':          { government: 'government_dubai',    private: 'private_dubai'    },
    'Sharjah':        { government: 'government_northern', private: 'private_sharjah'  },
    'Ajman':          { government: 'government_northern', private: 'private_northern' },
    'Ras Al Khaimah': { government: 'government_northern', private: 'private_northern' },
    'Fujairah':       { government: 'government_northern', private: 'private_northern' },
    'Umm Al Quwain':  { government: 'government_northern', private: 'private_northern' },
  };
  const key = emirateMap[emirate]?.[schoolType] || 'government_abudhabi';
  return EDUCATION_AUTHORITIES_DATA[key];
}

function renderAuthorityChannels() {
  if (!currentAuthority) return;
  const c = currentAuthority.channels;
  const box = document.getElementById('authority-channels');
  if (!box) return;

  const channelCards = [
    c.email_general  && { icon:'📧', label:'البريد العام',   val:c.email_general,   type:'email' },
    c.email_staffing && { icon:'👥', label:'شؤون الكادر',    val:c.email_staffing,  type:'email' },
    c.email_supply   && { icon:'🔄', label:'معلمو الاحتياط', val:c.email_supply,    type:'email' },
    c.email_vacancies&& { icon:'📋', label:'النواقص',        val:c.email_vacancies, type:'email' },
    c.nafis_email    && { icon:'🏆', label:'بريد Nafis',     val:c.nafis_email,     type:'email' },
    c.phone          && { icon:'📞', label:'الهاتف',         val:c.phone,           type:'phone' },
    c.eservices      && { icon:'🔗', label:'البوابة الإلكترونية', val:c.eservices,   type:'url'   },
    c.nafis_portal   && { icon:'🌐', label:'بوابة Nafis',    val:c.nafis_portal,    type:'url'   },
  ].filter(Boolean);

  box.innerHTML = channelCards.map(ch => {
    let link = '';
    if (ch.type === 'email') link = `href="mailto:${ch.val}"`;
    else if (ch.type === 'phone') link = `href="tel:${ch.val.replace(/\s/g,'')}"`;
    else link = `href="${ch.val}" target="_blank"`;
    return `
      
        
    
          
    ${ch.icon}
    
          
    ${ch.label}
    
          
    ${ch.val}
    
        
    
          `;
  }).join('');
}

function onReqTypeChange() {
  const sel = document.getElementById('req-type-select');
  const detailsArea = document.getElementById('req-details-area');
  const slaBox = document.getElementById('req-sla-box');
  const slaVal = document.getElementById('req-sla-val');
  const noteBox = document.getElementById('req-note-box');
  const docsList = document.getElementById('req-docs-list');

  if (!sel.value || !currentAuthority) { detailsArea.style.display='none'; slaBox.style.display='none'; return; }

  const rt = currentAuthority.request_types[sel.value];
  if (!rt) return;

  detailsArea.style.display = 'block';
  if (rt.sla_days) {
    slaBox.style.display = 'block';
    slaVal.textContent = `${rt.sla_days} يوم عمل`;
  } else {
    slaBox.style.display = 'none';
  }

  docsList.innerHTML = rt.required_docs.map(d => `
  • ${d}
    • `).join('');

  if (rt.note_ar) {
    noteBox.style.display = 'block';
    noteBox.textContent = `💡 ${rt.note_ar}`;
  } else {
    noteBox.style.display = 'none';
  }

  // تحديث placeholder
  const posInput = document.getElementById('req-position');
  const labelMap = {
    permanent_hire: 'مثال: معلم رياضيات — الصف 7',
    supply_teacher: 'مثال: معلم لغة عربية',
    vacancy_report: 'مثال: مرشد طلابي — المرحلة الإعدادية',
    staff_license:  'الاسم الكامل للموظف',
    staff_transfer: 'الاسم الكامل للموظف',
    nafis_registration: 'الاسم الكامل للموظف الإماراتي',
    emiratization_report: 'الربع / الفصل المُبلَّغ عنه',
  };
  if (posInput) posInput.placeholder = labelMap[sel.value] || 'تفاصيل المنصب';
}

}

function copyEmailText() {
  if (!currentEmailDraft) return;
  const text = `إلى: ${currentEmailDraft.to}\nالموضوع: ${currentEmailDraft.subject}\n\n${currentEmailDraft.body}`;
  navigator.clipboard.writeText(text).then(() => showToast('تم نسخ البريد الرسمي ✓'));
}

function openEmailClient() {
  if (!currentEmailDraft) return;
  const url = `mailto:${currentEmailDraft.to}?subject=${encodeURIComponent(currentEmailDraft.subject)}&body=${encodeURIComponent(currentEmailDraft.body)}`;
  window.open(url, '_blank');
}

function openAuthorityPortal() {
  if (!currentAuthority) return;
  const url = currentAuthority.channels.eservices || currentAuthority.channels.portal;
  if (url) window.open(url, '_blank');
}

async function saveOfficialRequest() {
  const sel = document.getElementById('req-type-select');
  if (!sel.value) { showToast('اختر نوع الطلب أولاً','amber'); return; }
  const position = document.getElementById('req-position')?.value || '';
  const dates    = document.getElementById('req-dates')?.value    || '';

  try {
    const resp = await sbPost('emiratization_requests', {
      request_type: sel.value,
      position_needed: position,
      notes: `${dates} | ${document.getElementById('req-extra')?.value||''}`.trim(),
      status: 'draft',
      academic_year: '2025-2026'
    });
    if (resp.ok || resp.status===201) {
      showToast('✅ تم حفظ الطلب في السجل');
      await loadCommsLog();
    } else {
      showToast('تعذّر الحفظ — جرب مجدداً','red');
    }
  } catch(e) {
    showToast('خطأ في الاتصال','red');
  }
}

async function saveAndClose() {
  await saveOfficialRequest();
  closeEmailModal();
}

async function loadCommsLog() {
  try {
    const resp = await fetch(`${SBU}/rest/v1/emiratization_requests?select=*&order=created_at.desc&limit=10`, {
      headers: SBH
    });
    const data = resp.ok ? await resp.json() : [];
    const box = document.getElementById('comms-log-list');
    if (!box) return;
    if (!data.length) {
      box.innerHTML = '
    لا توجد طلبات مسجّلة بعد
    ';
      return;
    }
    const typeLabels = {
      permanent_hire:'تعيين دائم',supply_teacher:'معلم احتياط',vacancy_report:'إبلاغ شغور',
      staff_license:'ترخيص موظف',staff_transfer:'نقل موظف',nafis_registration:'تسجيل Nafis',emiratization_report:'تقرير توطين'
    };
    const statusColors = { draft:'var(--text2)', sent:'#22d3ee', responded:'var(--emerald)', closed:'var(--text2)' };
    const statusLabels = { draft:'مسودة', sent:'مُرسَل', responded:'تم الرد', closed:'مُغلق' };
    box.innerHTML = data.map(r => `
      
    
        
    
          
    ${typeLabels[r.request_type]||r.request_type}${r.position_needed?' — '+r.position_needed:''}
    
          
    ${r.notes||''}
    
        
    
        
    
          
    ${statusLabels[r.status]||r.status}
    
          
    ${new Date(r.created_at).toLocaleDateString('ar-AE')}
    
        
    
      
    `).join('');
  } catch(e) { console.warn('comms log:', e); }
}

function showToast(msg, color='green'){
  const t=document.createElement('div');
  t.style.cssText=`position:fixed;bottom:80px;left:50%;transform:translateX(-50%);
    background:var(--card);border:1px solid ${color==='red'?'var(--red)':color==='amber'?'var(--amber)':'var(--emerald)'};
    color:var(--text);padding:10px 20px;border-radius:10px;font-family:'Tajawal',sans-serif;
    font-size:13px;font-weight:700;z-index:9999;box-shadow:0 4px 20px rgba(0,0,0,0.4);`;
  t.textContent=msg;
  document.body.appendChild(t);
  setTimeout(()=>t.remove(),3000);
}
